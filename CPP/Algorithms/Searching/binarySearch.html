Code for Binary Search Algorithm in C++

int binary_search(int arr[], int l, int r, int val)
{
int mid;
while (l <= r)
{
mid = (l + r) / 2;
if (arr[mid] == val)
return mid;
else if (arr[mid] < val)
return binary_search(arr, mid + 1, r, val);
else
return binary_search(arr, l, mid - 1, val);
}
return -1;
}
